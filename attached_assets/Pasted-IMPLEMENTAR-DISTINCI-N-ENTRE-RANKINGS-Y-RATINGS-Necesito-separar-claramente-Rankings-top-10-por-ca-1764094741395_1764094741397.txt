```typescript
import { collection, doc, getDoc, setDoc, getDocs, query, where, Timestamp } from 'firebase/firestore';
import { db } from './firebase';

export interface RankingEntry {
  position: number;
  userId: string;
  memberNumber: string;
  name: string;
  club: string;
  rating: number;
  tournamentsPlayed: number;
  wins: number;
  losses: number;
  photoURL?: string;
  lastChange: number;
  trend: 'up' | 'down' | 'stable' | 'new';
}

export interface Rankings {
  category: string;
  lastUpdated: Date;
  top10: RankingEntry[];
}

// Obtener rankings de una categor√≠a
export async function getRankings(category: string): Promise<RankingEntry[]> {
  const docRef = doc(db, 'rankings', category);
  const docSnap = await getDoc(docRef);
  
  if (docSnap.exists()) {
    const data = docSnap.data();
    return data.top10 || [];
  }
  
  return [];
}

// Recalcular rankings despu√©s de finalizar torneo
export async function recalculateRankings(tournamentType: string): Promise<void> {
  // Obtener todos los usuarios con sus ratings actualizados
  const usersSnapshot = await getDocs(collection(db, 'users'));
  const users = usersSnapshot.docs.map(doc => ({
    uid: doc.id,
    ...doc.data()
  }));
  
  // Filtrar por rol jugador y ordenar por rating
  const players = users
    .filter(u => u.role === 'jugador')
    .sort((a, b) => (b.profile?.rating || 0) - (a.profile?.rating || 0));
  
  // Obtener rankings anteriores para comparar cambios
  const previousRankings = await getRankings(tournamentType);
  
  // Crear top 10
  const top10: RankingEntry[] = players.slice(0, 10).map((player, index) => {
    const position = index + 1;
    const previousEntry = previousRankings.find(r => r.userId === player.uid);
    const previousPosition = previousEntry?.position || 0;
    
    let trend: 'up' | 'down' | 'stable' | 'new' = 'stable';
    let lastChange = 0;
    
    if (!previousEntry) {
      trend = 'new';
    } else {
      lastChange = previousPosition - position;
      if (lastChange > 0) trend = 'up';
      else if (lastChange < 0) trend = 'down';
    }
    
    return {
      position,
      userId: player.uid,
      memberNumber: player.memberNumber,
      name: `${player.profile?.firstName} ${player.profile?.lastName}`,
      club: player.profile?.club || '',
      rating: player.profile?.rating || 1000,
      tournamentsPlayed: player.profile?.tournamentsPlayed || 0,
      wins: player.profile?.wins || 0,
      losses: player.profile?.losses || 0,
      photoURL: player.profile?.photoURL,
      lastChange,
      trend
    };
  });
  
  // Guardar en Firestore
  await setDoc(doc(db, 'rankings', tournamentType), {
    category: tournamentType,
    lastUpdated: Timestamp.now(),
    top10
  });
}
```

```typescript
export async function getAllUsers() {
  const snapshot = await getDocs(collection(db, 'users'));
```
